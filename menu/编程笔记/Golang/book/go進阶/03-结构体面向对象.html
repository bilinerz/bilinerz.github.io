结构体面向对象

通过结构体嵌套和匿名字段来实现面向对象

    //定义父结构体
    type Person struct {
        name string
        age int
    }

    //定义子结构体
    type Student struct {
        Person //通过匿名字段继承父结构体
        school string
    }

    func main() {
        zhangSan := Student{
            Person{
                name:"张三",
                age:18,
            },
            school:"XX大学",
        }
        fmt.Println(zhangSan)

        //获取对应数据
        fmt.Println(zhangSan.name, zhangSan.age, zhangSan.school)
    }
