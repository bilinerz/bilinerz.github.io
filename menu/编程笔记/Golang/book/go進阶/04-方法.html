方法

方法也是函数，只是方法中还有接收者

语法：func (接收者) 方法名(参数)(返回值){ }

    //定义一个人的结构体
    type Person struct {
        name string
        age int
        sex string
    }

    func main() {
        //调用结构体
        zhangSan := Person{name:"张三", age:18, sex:"男"}
        //调用结构体下的work方法
        zhangSan.work()

        //取地址
        liSi := &Person{name:"李四", age:20, sex:"男"}

    }

    //定义一个work方法
    func (p Person) work() {
        fmt.Println(p.name, "正在工作...")
    }

    //定义一个rest方法，取的是地址
    func (p1 *Person) rest() {
        fmt.Println(p1.name, "正在休息...")
    }
