type 关键字

type关键字在go语法里是非常重要且常用的关键字，主要用于定义数据类型和结构体

定义新类型的语法：type 类型名 数据类型

    //定义一个新的int类型
    type myint int
    //定义一个新的string类型
    type mystr string

    func main() {
        var i1 myint
        i1 = 10
        i2 := 100
        fmt.Printf("i1的数据类型：%T，i2的数据类型：%T", i1, i2)

        var s1 mystr
        s1 = "张三"
        s2 := "李四"
        fmt.Printf("s1的数据类型：%T，s2的数据类型：%T", s1, s2)

        res1 := f1()
        fmt.Println(res1(10, 20))
    }

    //新定义函数类型
    type myfunc func(int, int)(string)

    //定义一个函数f1 将新定义的函数类型myfunc作为f1函数的返回值
    func f1() myfunc {
        fun := func(a, b int) string {
            s := strconv.Itoa(a) + strconv.Itoa(b)
            return s
        }
        return fun
    }

    //定义一个int类型的别名，相当于给int类型起小名，是可以与int类型通用的
    type myint2 = int
    i3 := 100
    myint2 = i3
    fmt.println(myint2)
