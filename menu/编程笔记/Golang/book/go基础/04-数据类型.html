    • 布尔类型

    package main

import "fmt" 

func main(){
    //布尔类型
    var b1 bool
    b1 = true
    fmt.Printf("b1的类型：%T，值为：%t\n", b1 , b1)
    //运行结果 b1的类型：bool，值为：true

    b2 := false
    fmt.Printf("b2的类型：%T，值为：%t\n", b2 , b2)
    //运行结果 b2的类型：bool，值为：false

}

    数值类型

    整数型

      有符号：最高位表示符号位，0整数，1负数，其余位表示数值

    • int8

      有符号 8 位整型，取值范围(-128 到 127)

      长度：8bit

    • int16

      有符号 16 位整型，取值范围(-32768 到 32767)

    • int32

      有符号 32 位整型，取值范围(-2147483648 到 2147483647)

    • int64

      有符号 64 位整型，取值范围(-9223372036854775808 到 9223372036854775807)

      • 无符号：所有的位表示数值

    • uint8

      无符号 8 位整型，取值范围(0 到 255)

      8位都用于表示数值

    • uint16

      无符号 16 位整型，取值范围(0 到 65535)

    • uint32

      无符号 32 位整型，取值范围(0 到 4294967295)

    • uint64

      无符号 64 位整型，取值范围(0 到 18446744073709551615)

    • byte：相对于 uint8

      rune：相当于 int32

    int和uint：根据底层平台，表示32或64位整数。除非需要使用特定大小的整数，否则通常使用int来表示整数。

        //整数
    var i1 int8
    i1 = 100
    fmt.Println(i1)
    var i2 uint8
    i2 = 200
    fmt.Println(i2)

    var i3 int
    i3 = 1000
    fmt.Println(i3)

    var (
        i4 uint8
        i5 byte
    )
    i4 = 100
    i5 = i4
    fmt.Println(i4, i5)

    //不指定类型，那么默认是 int
    var i6 = 100
    fmt.Printf("%T,%d\n", i6, i6)

    浮点类型

    • float32

      IEEE-754 32位浮点型数

    • float64

      IEEE-754 64位浮点型数

        //浮点
    var f1 float32
    f1 = 3.14
    var f2 float64
    f2 = 4.67
    fmt.Printf("%T,%f\n", f1, f1) //打印输出默认保留小数点后6位
    fmt.Printf("%T,%.2f\n", f1, f1) //打印输保留小数点后2位
    fmt.Printf("%T,%f\n", f2, f2)

    字符串

    多个byte的集合，也可理解为一个字符序列

        //定义字符串
    var s1 string
    s1 = "张三"
    fmt.Printf("%T, %s\n", s1, s1)

    s2 := `Hello World`
    fmt.Printf("%T, %s\n", s2, s2)

    //单引号'' 和 双引号"" 的区别
    v1 := 'A'
    v2 := "A"
    fmt.Printf("%T, %d\n", v1, v1) //打印出 int32 类型
    fmt.Printf("%T, %s\n", v2, v2) //打印出 string 类型

    数据类型转换

    go语言在定义、赋值和运算时数据类型必须一致

    在转换数据类型时只有兼容类型可以转换

    常量：在有需要的时候自动转换

    变量：需要手动转换

        var a int8
    a = 10

    var b int16
    b = int16(a)  //将a转换为int16才能赋值给b

    fmt.Println(a, b)

    f1 := 3.14
    var c int
    c = int(f1)  //将f1转换为int才能赋值给c
    fmt.Println(f1, c)
