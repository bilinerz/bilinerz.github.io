strings包的使用

主要用于字符串的操作

    str := "Hello World"

    //Contains 关键字：可以用来判断字符串中是否存在指定字符串
    fmt.Println(strings.Contains(str, "Hello"))  //如果str中存在Hello字符串就返回true否则返回false
    
    //Count：可以用来统计字符串中指定字符出现的次数
    fmt.Println(strings.Count(str, "l"))  //返回 str 中 l 的个数

    //Index：获取指定字符串第一次出现的下标，如果字符不存在则返回 -1
    fmt.Println(strings.Index(str, "e"))

    //LastIndex：获取指定字符最后一次出现的下标
    fmt.Println(strings.LastIndex(str, "l"))

    //ToLower：将所有字母转为小写
    fmt.Println(strings.ToLower(str))

    //ToUpper：将所有字母转为大写
    fmt.Println(strings.ToUpper(str))

    //Repeat：重复将自己拼接自己
    str0 := strings.Repeat("hello", 5)
    fmt.Println(str0)

    str1 := "20230817笔记.txt"

    //HasPrefix：用来判断字符串开头的字符
    if strings.HasPrefix(str1, "2023"){
        fmt.Println("这是2023年的文件")
    }

    //HasSuffix：用来判断字符串末尾的字符
    if strings.HasSuffix(str1, ".txt") {
        fmt.Println("这是txt文档")
    }

    strArr := []string{"张三", "李四", "王五"}

    //Join：以指定字符将切片拼接成字符串
    str2 := strings.Join(strArr, "-")  //以符号 - 来拼接切片中的元素成新的字符串
    fmt.Println(str2)

    str3 := "123,abc,456,xyz,789"
    //Split：以指定字符将字符串切割成切片
    str4 := strings.Split(str3, ",")  //以符号 , 将字符串切割成切片
    fmt.Println(str4)

    //Replace：将自定字符替换为另一个指定字符
    str5 := strings.Replace(str3, ",", "-", -1)  //将 , 替换为 - 最后的 -1 表示全部替换，输入1就替换1个
    fmt.Println(str5)
