数组的遍历

    arr1 := [5]int{1, 2, 3, 4, 5}

    //使用for循环遍历
    for i := 0; i < len(arr1); i++ {
        fmt.Println(arr1[i])
    }

    fmt.Println("---------------")

    //使用 for range 遍历
    for index, value := range arr1 {
        fmt.Printf("下标为:%d，数值是:%d\n", index, value)
    }
